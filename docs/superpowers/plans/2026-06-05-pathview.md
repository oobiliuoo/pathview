# PathView 实施计划

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** 构建一个机器人路径可视化 Web 应用，支持 CSV 导入、3D 路径渲染、姿态显示、回放动画和多路径对比。

**Architecture:** React SPA 前端通过 REST API 与 Express 后端通信。后端解析 CSV、管理路径数据、SQLite 持久化。前端用 react-three-fiber 渲染 3D 场景。

**Tech Stack:** React 18 + Vite + TypeScript + @react-three/fiber + @react-three/drei | Express + TypeScript + better-sqlite3 + multer + csv-parse

`server/src/services/csvParser.ts`:

```typescript
import { parse } from 'csv-parse';
import { createReadStream } from 'fs';

export interface ParsedCsv {
  columns: string[];
  rows: Record<string, string | number>[];
  rowCount: number;
}

export function parseCsv(filePath: string): Promise<ParsedCsv> {
  return new Promise((resolve, reject) => {
    const rows: Record<string, string | number>[] = [];
    let columns: string[] = [];

    createReadStream(filePath, { encoding: 'utf-8' })
      .pipe(parse({ columns: true, skip_empty_lines: true, relax_column_count: true }))
      .on('headers', (headers: string[]) => {
        columns = headers.map(h => h.trim());
      })
      .on('data', (row: Record<string, string>) => {
        const numericRow: Record<string, string | number> = {};
        for (const [key, val] of Object.entries(row)) {
          const trimmedKey = key.trim();
          const num = Number(val?.trim());
          numericRow[trimmedKey] = isNaN(num) ? (val?.trim() ?? '') : num;
        }
        rows.push(numericRow);
      })
      .on('end', () => {
        resolve({ columns, rows, rowCount: rows.length });
      })
      .on('error', (err: Error) => {
        reject(new Error(`CSV parse error at row ${rows.length + 1}: ${err.message}`));
      });
  });
}

export function getPreviewRows(rows: Record<string, string | number>[], count = 3): Record<string, string | number>[] {
  return rows.slice(0, count);
}
```

- [ ] **Step 2: 创建 columnMatcher.ts**

`server/src/services/columnMatcher.ts`:

```typescript
const FIELD_KEYWORDS: Record<string, string[]> = {
  x: ['x', 'pos_x', 'px', 'position_x'],
  y: ['y', 'pos_y', 'py', 'position_y'],
  z: ['z', 'pos_z', 'pz', 'position_z'],
  rx: ['rx', 'rot_x', 'roll'],
  ry: ['ry', 'rot_y', 'pitch'],
  rz: ['rz', 'rot_z', 'yaw'],
};

export function suggestMapping(columns: string[]): Record<string, string | null> {
  const mapping: Record<string, string | null> = {};
  const usedColumns = new Set<string>();

  // Position fields first (required)
  for (const field of ['x', 'y', 'z']) {
    mapping[field] = findMatch(columns, FIELD_KEYWORDS[field], usedColumns);
  }

  // Orientation fields (optional)
  for (const field of ['rx', 'ry', 'rz']) {
    mapping[field] = findMatch(columns, FIELD_KEYWORDS[field], usedColumns);
  }

  return mapping;
}

function findMatch(columns: string[], keywords: string[], usedColumns: Set<string>): string | null {
  const lowerColumns = columns.map(c => c.toLowerCase().replace(/[^a-z0-9_]/g, ''));

  for (const keyword of keywords) {
    for (let i = 0; i < columns.length; i++) {
      if (usedColumns.has(columns[i])) continue;
      if (lowerColumns[i] === keyword || lowerColumns[i].endsWith('_' + keyword) || lowerColumns[i].startsWith(keyword + '_')) {
        usedColumns.add(columns[i]);
        return columns[i];
      }
    }
  }

  return null;
}
```

- [ ] **Step 1: 创建路径路由**

`server/src/routes/paths.ts`:

```typescript
import { Router } from 'express';
import multer from 'multer';
import { join, dirname } from 'path';
import { fileURLToPath } from 'url';
import { randomUUID } from 'uuid';
import { readFileSync, readdirSync, unlinkSync } from 'fs';
import db from '../db.js';
import { parseCsv, getPreviewRows } from '../services/csvParser.js';
import { suggestMapping } from '../services/columnMatcher.js';
import type { CreatePathRequest, CsvUploadResult } from '../types/index.js';

const __dirname = dirname(fileURLToPath(import.meta.url));
const uploadsDir = join(__dirname, '../../../uploads');

const upload = multer({
  dest: uploadsDir,
  limits: { fileSize: 50 * 1024 * 1024 },
});

const router = Router();

const PATH_COLORS = [
  '#3b82f6', '#f59e0b', '#22c55e', '#ef4444',
  '#a855f7', '#ec4899', '#06b6d4', '#f97316',
];

// Upload CSV and get column info
router.post('/upload', upload.single('file'), async (req, res) => {
  if (!req.file) {
    res.status(400).json({ error: 'No file uploaded' });
    return;
  }

  try {
    const { columns, rows, rowCount } = await parseCsv(req.file.path);
    const preview = getPreviewRows(rows);
    const suggestedMapping = suggestMapping(columns);

    // Check x/y/z are all mapped
    if (!suggestedMapping.x || !suggestedMapping.y || !suggestedMapping.z) {
      res.status(400).json({
        error: 'Could not auto-detect position columns (X/Y/Z). Please check your CSV.',
        columns,
        preview,
        suggestedMapping,
        rowCount,
      });
      return;
    }

    const result: CsvUploadResult = {
      fileId: req.file.filename,
      columns,
      preview,
      suggestedMapping,
      rowCount,
    };

    res.json(result);
  } catch (err) {
    res.status(500).json({ error: (err as Error).message });
  }
});

// Create path with column mapping
router.post('/', (req, res) => {
  const { name, sourceFile, mapping, fileId } = req.body as CreatePathRequest;

  if (!name || !mapping || !fileId) {
    res.status(400).json({ error: 'Missing required fields: name, mapping, fileId' });
    return;
  }

  if (!mapping.x || !mapping.y || !mapping.z) {
    res.status(400).json({ error: 'X, Y, Z mappings are required' });
    return;
  }

  const filePath = join(uploadsDir, fileId);
  let rows: Record<string, string | number>[];

  try {
    const parsed = readFileSync(filePath, 'utf-8');
    // Re-parse synchronously for creation
    const { parse } = await import('csv-parse/sync');
    rows = parse(parsed, { columns: true, skip_empty_lines: true, relax_column_count: true });
  } catch {
    res.status(400).json({ error: 'Uploaded file not found or expired. Please re-upload.' });
    return;
  }

  const hasOrientation = !!(mapping.rx || mapping.ry || mapping.rz);

  // Assign color
  const existingCount = (db.prepare('SELECT COUNT(*) as count FROM paths').get() as { count: number }).count;
  const color = PATH_COLORS[existingCount % PATH_COLORS.length];

  const createPath = db.transaction(() => {
    const insertPath = db.prepare(`
      INSERT INTO paths (name, source_file, has_orientation, point_count, color)
      VALUES (?, ?, ?, ?, ?)
    `);
    const result = insertPath.run(name, sourceFile || null, hasOrientation ? 1 : 0, rows.length, color);
    const pathId = result.lastInsertRowid as number;

    const insertPoint = db.prepare(`
      INSERT INTO path_points (path_id, seq, x, y, z, rx, ry, rz)
      VALUES (?, ?, ?, ?, ?, ?, ?, ?)
    `);
    const insertMapping = db.prepare(`
      INSERT INTO column_mappings (path_id, field, csv_column)
      VALUES (?, ?, ?)
    `);

    for (let i = 0; i < rows.length; i++) {
      const row = rows[i];
      const x = Number(row[mapping.x]) || 0;
      const y = Number(row[mapping.y]) || 0;
      const z = Number(row[mapping.z]) || 0;
      const rx = mapping.rx ? (Number(row[mapping.rx]) || null) : null;
      const ry = mapping.ry ? (Number(row[mapping.ry]) || null) : null;
      const rz = mapping.rz ? (Number(row[mapping.rz]) || null) : null;
      insertPoint.run(pathId, i, x, y, z, rx, ry, rz);
    }

    for (const [field, csvCol] of Object.entries(mapping)) {
      if (csvCol) {
        insertMapping.run(pathId, field, csvCol);
      }
    }

    return pathId;
  });

  try {
    const pathId = createPath();

    // Clean up uploaded file
    try { unlinkSync(filePath); } catch {}

    const path = db.prepare('SELECT * FROM paths WHERE id = ?').get(pathId);
    res.status(201).json(path);
  } catch (err) {
    res.status(500).json({ error: (err as Error).message });
  }
});

// GET all paths
router.get('/', (_req, res) => {
  const paths = db.prepare('SELECT * FROM paths ORDER BY created_at DESC').all();
  res.json(paths);
});

// GET path by id with points
router.get('/:id', (req, res) => {
  const path = db.prepare('SELECT * FROM paths WHERE id = ?').get(req.params.id);
  if (!path) {
    res.status(404).json({ error: 'Path not found' });
    return;
  }
  const points = db.prepare('SELECT * FROM path_points WHERE path_id = ? ORDER BY seq').all(req.params.id);
  res.json({ ...path, points });
});

// DELETE path
router.delete('/:id', (req, res) => {
  const result = db.prepare('DELETE FROM paths WHERE id = ?').run(req.params.id);
  if (result.changes === 0) {
    res.status(404).json({ error: 'Path not found' });
    return;
  }
  res.status(204).send();
});

// GET path points with pagination
router.get('/:id/points', (req, res) => {
  const limit = Math.min(parseInt(req.query.limit as string) || 1000, 10000);
  const page = Math.max(parseInt(req.query.page as string) || 1, 1);
  const offset = (page - 1) * limit;

  const points = db.prepare(
    'SELECT * FROM path_points WHERE path_id = ? ORDER BY seq LIMIT ? OFFSET ?'
  ).all(req.params.id, limit, offset);

  const total = (db.prepare('SELECT COUNT(*) as count FROM path_points WHERE path_id = ?').get(req.params.id) as { count: number }).count;

  res.json({ points, page, limit, total });
});

export default router;
```

```typescript
import { useState, useEffect, useCallback } from 'react';
import * as api from '../api/paths';
import type { PathData, PathWithPoints } from '../types';

export function usePaths() {
  const [paths, setPaths] = useState<PathData[]>([]);
  const [selectedPathId, setSelectedPathId] = useState<number | null>(null);
  const [pathsWithPoints, setPathsWithPoints] = useState<Map<number, PathWithPoints>>(new Map());
  const [loading, setLoading] = useState(false);

  const fetchPaths = useCallback(async () => {
    setLoading(true);
    try {
      const data = await api.getPaths();
      setPaths(data);
    } finally {
      setLoading(false);
    }
  }, []);

  useEffect(() => { fetchPaths(); }, [fetchPaths]);

  const loadPathPoints = useCallback(async (id: number) => {
    if (pathsWithPoints.has(id)) return;
    const data = await api.getPathWithPoints(id);
    setPathsWithPoints(prev => new Map(prev).set(id, data));
  }, [pathsWithPoints]);

  const removePath = useCallback(async (id: number) => {
    await api.deletePath(id);
    setPaths(prev => prev.filter(p => p.id !== id));
    setPathsWithPoints(prev => {
      const next = new Map(prev);
      next.delete(id);
      return next;
    });
    if (selectedPathId === id) setSelectedPathId(null);
  }, [selectedPathId]);

  const activePaths = Array.from(pathsWithPoints.values());

  return {
    paths,
    activePaths,
    selectedPathId,
    setSelectedPathId,
    loading,
    loadPathPoints,
    removePath,
    fetchPaths,
  };
}
```

`client/src/hooks/usePlayback.ts`:

```typescript
import { useState, useRef, useCallback, useEffect } from 'react';
import type { PlaybackMode, PlaybackState } from '../types';

interface UsePlaybackOptions {
  totalPoints: number;
  onIndexChange?: (index: number) => void;
}

export function usePlayback({ totalPoints, onIndexChange }: UsePlaybackOptions) {
  const [state, setState] = useState<PlaybackState>({
    isPlaying: false,
    currentIndex: 0,
    speed: 1,
    mode: 'step',
    followCamera: false,
  });

  const timerRef = useRef<ReturnType<typeof setTimeout> | null>(null);
  const onIndexChangeRef = useRef(onIndexChange);
  onIndexChangeRef.current = onIndexChange;

  const clearTimer = useCallback(() => {
    if (timerRef.current) {
      clearTimeout(timerRef.current);
      timerRef.current = null;
    }
  }, []);

  const advance = useCallback(() => {
    setState(prev => {
      const nextIndex = prev.currentIndex + 1;
      if (nextIndex >= totalPoints) {
        if (prev.mode === 'loop') {
          onIndexChangeRef.current?.(0);
          return { ...prev, currentIndex: 0 };
        }
        return { ...prev, isPlaying: false, currentIndex: prev.currentIndex };
      }
      onIndexChangeRef.current?.(nextIndex);
      return { ...prev, currentIndex: nextIndex };
    });
  }, [totalPoints]);

  useEffect(() => {
    if (state.isPlaying && totalPoints > 0) {
      const interval = state.mode === 'step'
        ? 500 / state.speed
        : 50 / state.speed;

      const tick = () => {
        advance();
        timerRef.current = setTimeout(tick, interval);
      };
      timerRef.current = setTimeout(tick, interval);
    }

    return clearTimer;
  }, [state.isPlaying, state.speed, state.mode, advance, clearTimer]);

  const play = useCallback(() => {
    if (state.currentIndex >= totalPoints - 1 && state.mode !== 'loop') {
      setState(prev => ({ ...prev, currentIndex: 0, isPlaying: true }));
      onIndexChangeRef.current?.(0);
    } else {
      setState(prev => ({ ...prev, isPlaying: true }));
    }
  }, [state.currentIndex, state.mode, totalPoints]);

  const pause = useCallback(() => {
    setState(prev => ({ ...prev, isPlaying: false }));
    clearTimer();
  }, [clearTimer]);

  const stop = useCallback(() => {
    clearTimer();
    setState(prev => ({ ...prev, isPlaying: false, currentIndex: 0 }));
    onIndexChangeRef.current?.(0);
  }, [clearTimer]);

  const goTo = useCallback((index: number) => {
    const clamped = Math.max(0, Math.min(index, totalPoints - 1));
    setState(prev => ({ ...prev, currentIndex: clamped }));
    onIndexChangeRef.current?.(clamped);
  }, [totalPoints]);

  const stepForward = useCallback(() => {
    goTo(state.currentIndex + 1);
  }, [state.currentIndex, goTo]);

  const stepBackward = useCallback(() => {
    goTo(state.currentIndex - 1);
  }, [state.currentIndex, goTo]);

  const setSpeed = useCallback((speed: number) => {
    setState(prev => ({ ...prev, speed }));
  }, []);

  const setMode = useCallback((mode: PlaybackMode) => {
    setState(prev => ({ ...prev, mode }));
  }, []);

  const setFollowCamera = useCallback((followCamera: boolean) => {
    setState(prev => ({ ...prev, followCamera }));
  }, []);

  return {
    state,
    play,
    pause,
    stop,
    goTo,
    stepForward,
    stepBackward,
    setSpeed,
    setMode,
    setFollowCamera,
  };
}
```

```tsx
import { useMemo } from 'react';
import * as THREE from 'three';
import { Line } from '@react-three/drei';
import type { PathPoint } from '../types';

interface OrientationAxesProps {
  points: PathPoint[];
  axisLength?: number;
}

const AXIS_COLORS = {
  x: '#ef4444',
  y: '#22c55e',
  z: '#3b82f6',
};

export function OrientationAxes({ points, axisLength = 5 }: OrientationAxesProps) {
  const axesData = useMemo(() => {
    return points
      .filter(p => p.rx != null && p.ry != null && p.rz != null)
      .map(point => {
        const euler = new THREE.Euler(
          THREE.MathUtils.degToRad(point.rx!),
          THREE.MathUtils.degToRad(point.ry!),
          THREE.MathUtils.degToRad(point.rz!),
          'ZYX'
        );

        const origin = new THREE.Vector3(point.x, point.y, point.z);
        const xDir = new THREE.Vector3(1, 0, 0).applyEuler(euler).multiplyScalar(axisLength);
        const yDir = new THREE.Vector3(0, 1, 0).applyEuler(euler).multiplyScalar(axisLength);
        const zDir = new THREE.Vector3(0, 0, 1).applyEuler(euler).multiplyScalar(axisLength);

        return {
          origin,
          xEnd: origin.clone().add(xDir),
          yEnd: origin.clone().add(yDir),
          zEnd: origin.clone().add(zDir),
        };
      });
  }, [points, axisLength]);

  return (
    <group>
      {axesData.map((axes, i) => (
        <group key={i}>
          <Line points={[axes.origin, axes.xEnd]} color={AXIS_COLORS.x} lineWidth={1.5} />
          <Line points={[axes.origin, axes.yEnd]} color={AXIS_COLORS.y} lineWidth={1.5} />
          <Line points={[axes.origin, axes.zEnd]} color={AXIS_COLORS.z} lineWidth={1.5} />
        </group>
      ))}
    </group>
  );
}
```

```tsx
import { useState, useCallback } from 'react';
import { Scene3D } from './components/Scene3D';
import { PathList } from './components/PathList';
import { DisplayControls } from './components/DisplayControls';
import { PointDetail } from './components/PointDetail';
import { PlaybackBar } from './components/PlaybackBar';
import { ColumnMapper } from './components/ColumnMapper';
import { usePaths } from './hooks/usePaths';
import { usePlayback } from './hooks/usePlayback';
import * as api from './api/paths';
import type { PathPoint, DisplayOptions, CsvUploadResult } from './types';

export default function App() {
  const {
    paths, activePaths, selectedPathId, setSelectedPathId,
    loadPathPoints, removePath, fetchPaths,
  } = usePaths();

  const [visiblePaths, setVisiblePaths] = useState<Set<number>>(new Set());
  const [displayOptions, setDisplayOptions] = useState<DisplayOptions>({
    showOrientation: true,
    showPathLine: true,
    showPoints: true,
  });
  const [selectedPoint, setSelectedPoint] = useState<PathPoint | null>(null);
  const [uploadResult, setUploadResult] = useState<CsvUploadResult | null>(null);

  const selectedPath = activePaths.find(p => p.id === selectedPathId);
  const totalPoints = selectedPath?.points.length ?? 0;

  const playback = usePlayback({
    totalPoints,
    onIndexChange: (index) => {
      if (selectedPath) {
        setSelectedPoint(selectedPath.points[index] ?? null);
      }
    },
  });

  const displayedPaths = activePaths.filter(p => visiblePaths.has(p.id));

  const handleSelectPath = useCallback(async (id: number) => {
    setSelectedPathId(id);
    await loadPathPoints(id);
    setVisiblePaths(prev => new Set(prev).add(id));
  }, [loadPathPoints, setSelectedPathId]);

  const handleImportCsv = useCallback(async (file: File) => {
    try {
      const result = await api.uploadCsv(file);
      setUploadResult(result);
    } catch (err) {
      alert((err as Error).message);
    }
  }, []);

  const handleMappingConfirm = useCallback(async (mapping: Record<string, string | null>, name: string) => {
    if (!uploadResult) return;
    try {
      await api.createPath({
        name: name || uploadResult.fileId,
        sourceFile: file?.name || '',
        mapping,
        fileId: uploadResult.fileId,
      });
      setUploadResult(null);
      await fetchPaths();
    } catch (err) {
      alert((err as Error).message);
    }
  }, [uploadResult, fetchPaths]);

  const handleToggleVisible = useCallback((id: number) => {
    setVisiblePaths(prev => {
      const next = new Set(prev);
      if (next.has(id)) next.delete(id);
      else next.add(id);
      return next;
    });
  }, []);

  return (
    <div style={{ display: 'flex', height: '100vh', width: '100vw' }}>
      {/* Left panel */}
      <div style={{ display: 'flex', flexDirection: 'column' }}>
        <PathList
          paths={paths}
          selectedPathId={selectedPathId}
          onSelectPath={handleSelectPath}
          onDeletePath={removePath}
          onImportCsv={handleImportCsv}
          visiblePaths={visiblePaths}
          onToggleVisible={handleToggleVisible}
        />
        <DisplayControls options={displayOptions} onChange={setDisplayOptions} />
      </div>

      {/* Center */}
      <div style={{ flex: 1, display: 'flex', flexDirection: 'column' }}>
        <div style={{ flex: 1 }}>
          <Scene3D
            paths={displayedPaths}
            displayOptions={displayOptions}
            currentPointIndex={playback.state.currentIndex}
            currentPathId={selectedPathId}
            onPointClick={setSelectedPoint}
          />
        </div>
        <PlaybackBar
          state={playback.state}
          totalPoints={totalPoints}
          onPlay={playback.play}
          onPause={playback.pause}
          onStop={playback.stop}
          onStepForward={playback.stepForward}
          onStepBackward={playback.stepBackward}
          onGoTo={playback.goTo}
          onSetSpeed={playback.setSpeed}
          onSetMode={playback.setMode}
          onSetFollowCamera={playback.setFollowCamera}
        />
      </div>

      {/* Right panel */}
      <PointDetail point={selectedPoint} />

      {/* Column mapper dialog */}
      {uploadResult && (
        <ColumnMapper
          uploadResult={uploadResult}
          onConfirm={handleMappingConfirm}
          onCancel={() => setUploadResult(null)}
        />
      )}
    </div>
  );
}
```

> **注意：** `handleMappingConfirm` 中引用了 `file` 变量，但它不在作用域内。需要在 `handleImportCsv` 中保存文件名。修正方法：在组件中增加 `const [importFileName, setImportFileName] = useState('')`，在 `handleImportCsv` 中设置它，在 `handleMappingConfirm` 中使用 `importFileName` 代替 `file?.name`。

```tsx
import { useMemo } from 'react';
// ... existing imports

// Add inside Scene3D component, before return:
const scaleFactor = useMemo(() => {
  if (paths.length === 0) return 1;
  let min = new THREE.Vector3(Infinity, Infinity, Infinity);
  let max = new THREE.Vector3(-Infinity, -Infinity, -Infinity);
  for (const path of paths) {
    for (const p of path.points) {
      min.min(new THREE.Vector3(p.x, p.y, p.z));
      max.max(new THREE.Vector3(p.x, p.y, p.z));
    }
  }
  const diagonal = min.distanceTo(max);
  return diagonal > 0 ? diagonal * 0.01 : 1;
}, [paths]);
```
